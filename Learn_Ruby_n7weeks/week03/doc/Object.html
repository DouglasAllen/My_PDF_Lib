def compare(x,y)
  if x.zero?
    1
  elsif y.zero?
    -1
  else x <=> y
  end
end

def filereadwrite(file1, file2)
  f2 = File.open(file2, "w")
  f1 = File.open(file1, "r")
  while line = f1.gets
    f2.puts line
  end
  f1.close
  f2.close
end

def mtdarry  
  10.times do |num|  
    square = num * num  
    return num, square if num > 5  
  end  
end
